period = 1

def get_indicators(df, period):
    price_change = df['Adj Close'].shift(-period) - df['Adj Close']
    df['Response'] = df['Adj Close'].shift(-period) # 실제 주가로 설정

    delta = df['Adj Close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up = up.rolling(14).mean()
    roll_down = down.abs().rolling(14).mean()
    RS = roll_up / roll_down
    df['RSI'] = 100.0 - (100.0 / (1.0 + RS))

    exp12     = df['Adj Close'].ewm(span=12, adjust=False).mean()
    exp26     = df['Adj Close'].ewm(span=26, adjust=False).mean()
    macd      = exp12 - exp26
    signal    = macd.ewm(span=9, adjust=False).mean()
    df['MACD'] = macd - signal

    df['PriceRateOfChange'] = df['Adj Close'].pct_change(periods = 14)

    df['OnBalanceVolume'] = (np.sign(df['Adj Close'].diff()) * df['Volume']).fillna(0).cumsum()

    # 이동평균선 추가
    df['MA5'] = df['Adj Close'].rolling(window=5).mean()
    df['MA20'] = df['Adj Close'].rolling(window=20).mean()

    df = df.dropna()
    df = df[:-period]
    return df
